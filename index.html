ข้อที่ 5: การตกแต่งปุ่มด้วย CSS

style.css
/* เขียน CSS ของปุ่ม .modern-button ที่นี่ */
.modern-button {
    background-color: #007bff; /* 1. ตั้งค่าสีพื้นหลังเป็นสีน้ำเงินเข้ม */
    color: #ffffff;            /* 2. ตั้งค่าสีตัวอักษรเป็นสีขาว */
    padding: 10px 20px;       /* 3. เพิ่มระยะห่างภายใน บน-ล่าง 10px และ ซ้าย-ขวา 20px */
    border-radius: 5px;        /* 4. ทำให้ขอบปุ่มมน 5px */
    border: none;              /* 5. ลบเส้นขอบเริ่มต้นของปุ่มออก */
    cursor: pointer;           /* 6. เปลี่ยนรูปแบเคอร์เซอร์เป็นรูปมือเมื่อนำเมาส์ไปชี้ */
    
    /* 8. เพิ่มการเปลี่ยนผ่าน (transition) ให้ดูนุ่มนวลในเวลา 0.3 วินาที */
    transition: background-color 0.3s ease;
    
    /* ตกแต่งเพิ่มเติมเพื่อให้ดูทันสมัยขึ้น */
    font-size: 16px;
    font-weight: bold;
}

/* 7. เพิ่มเอฟเฟกต์เมื่อนำเมาส์ไปชี้ (hover) */
.modern-button:hover {
    background-color: #0056b3; /* เปลี่ยนสีพื้นหลังให้เข้มขึ้น */
}

ข้อที่ 6: การตกแต่งการ์ดโปรไฟล์ด้วย CSS

style.css
.profile-card {
    /* 1. กำหนดความกว้างสูงสุด 300px และจัดให้อยู่กึ่งกลาง */
    max-width: 300px;
    margin: auto; 

    /* 2. เพิ่มระยะห่างภายใน (padding) 20px */
    padding: 20px;

    /* 3. เพิ่มขอบโค้ง (border-radius) 10px */
    border-radius: 10px;

    /* 4. เปลี่ยนสีพื้นหลังเป็นสีขาว (#ffffff) */
    background-color: #ffffff;

    /* 5. เพิ่มเงา (box-shadow) ให้การ์ดดูมีมิติ */
    box-shadow: 0 4px 8px rgba(0, 0, 0, 0.1);

    /* 7. จัดข้อความทั้งหมดภายในให้กึ่งกลาง */
    text-align: center;
